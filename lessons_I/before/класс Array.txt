#include <iostream> 
#include <cstdlib>

using namespace std;

class Array
{
private:
	int* mas;
	int size;
public:
	Array(int* arr, int n)
	{
		mas = arr;
		size = n;
	}
	void random()
	{
		for (int i = 0; i < size; i++)
			mas[i] = rand() % 20;
	}
	int findMax()
	{
		int max = mas[0];
		for (int i = 1; i < size; i++)
			if (mas[i] > max)
				max = mas[i];
		return max;
	}
	int findMin()
	{
		int min= mas[0];
		for (int i = 1; i < size; i++)
			if (mas[i] < min)
				min = mas[i];
		return min;
	}
	int findSum()
	{
		int res = 0;
		for (int i = 0; i < size; i++)
			res += mas[i];
		return res;
	}
	void print()
	{
		for (int i = 0; i < size; i++)
			cout << mas[i] << " ";
		cout << endl;
	}
	void sort()
	{
		for (int i = 0; i < size; i++)
		{
			int indx_min = i;
			for (int j = i + 1; j < size; j++)
				if (mas[j] < mas[indx_min])
					indx_min = j;
			int g = mas[indx_min];
			mas[indx_min] = mas[i];
			mas[i] = g;
		}
	}
};


int main()
{
	int massiv[5] = { 1, 4, 6, 2, 3 };
	Array one(massiv, 5);
	one.random();
	one.print();
	one.sort();
	one.print();
	system("pause");
	return 0;
}
