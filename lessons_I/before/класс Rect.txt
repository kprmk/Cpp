#include <iostream> 

using std::cout;
using std::endl;

class Rectangle
{
private:
	int a;
	int b;
public:
	Rectangle(int x, int y)
	{
		a = x;
		b = y;
	}
	Rectangle(int x)
	{
		a = b = x;
	}
	int findMaxSide()
	{
		if (a > b)
			return a;
		else
			return b;
	}
	int findSquare()
	{
		return a * b;
	}
	int findPerimetr()
	{
		return 2 * (a + b);
	}
	void draw()
	{
		for (int j = 0; j < b; j++)
		{
			for (int i = 0; i < a; i++)
				if (i == 0 || i == a - 1 || j == 0 || j == b-1)
					cout << "*";
				else
					cout << " ";
			cout << endl;
		}
	}
};

int main()
{
	Rectangle one(15, 6);
	one.draw();
	system("pause");
	return 0;
}
